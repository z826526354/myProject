## 响应式布局——计算viewport

利用 js 计算viewport的值，实现真正意义上的自适应

```html
<head>
	<meta name="viewport" content="width=device-width, initial-scale=1.0 user-scalable=no minmum-scale=1 maxmum-scale=1" id='view'>
    <script>
    	(function () {
            var curWidth = document.documentElement.clientWidth;
            var curWidth = window.innerWidth;
            var curWidth = window.screen.width;
            
            console.log(curWidth);
            
            var targetWidth = 375;
            var scale = curWidth / targetWidth;
            console.log(scale);
            
            var view = document.getElementById('view');
            console.log(view.content);
            
            view.content='initial-scale=' + scale +',user-scalable=no, minimum-scale='+ scale +', maxmum-scale='+ scale +''; 
            
        })()
    </script>
</head>
```
